Select	a.nome,
	a.sobrenome, 
	b.nota1,
	b.nota2,
	b.nota3 
from	alunos a, 
	boletim b
where	a.id_alunos = b.id_alunos;


Select	a.nome, 
	a.sobrenome, 
	round((b.nota1+b.nota2+b.nota3) / 3) as media
from	alunos a,
	boletim b
where a.id_alunos = b.id_alunos;
